      import numpy as np
      ​
      P = np.array([[0.1, 0.6, 0.3],
                    [0.4, 0.3, 0.3],
                    [0.2, 0.5, 0.3]])
      ​
      def simuler_chaine(P, etat_init, n):
          etats = [etat_init]
          for _ in range(n - 1):
              etat_courant = etats[-1]
              etats.append(np.random.choice([0, 1, 2], p=P[etat_courant]))
          return etats
      ​
      n_trajectoires = 10000
      n_transitions = 10
      comptes = np.zeros((n_transitions, 3))
      ​
      for _ in range(n_trajectoires):
          trajectoire = simuler_chaine(P, etat_init=0, n=n_transitions)
          for t, etat in enumerate(trajectoire):
              comptes[t, etat] += 1  
              
      probas_estimees = comptes / n_trajectoires
      ​
      for t in range(n_transitions):
          print(f"Après {t+1} transitions: {probas_estimees[t]}")

      import matplotlib.pyplot as plt
      ​
      n = 1000
      mu = 5
      sigma2 = 4
      epsilon = 2
      ​
      X = np.random.normal(loc=mu, scale=np.sqrt(sigma2), size=n)
      ​

      freq_empirique = np.mean(np.abs(X - mu) >= epsilon)

      def suite_harmonique(n):
          """Génère les n premiers termes de la suite harmonique."""
          u = []
          somme = 0
          for k in range(1, n+1):
              somme += 1 / k
              u.append(somme)
          return u
      ​
      def aitken(u):
          """Applique la méthode d’Aitken à une suite (liste) u."""
          u_prime = []
          for i in range(len(u) - 2):
              num = (u[i + 1] - u[i]) ** 2
              denom = u[i + 2] - 2 * u[i + 1] + u[i]
              if denom != 0:
                  u_acc = u[i] - num / denom
              else:
                  u_acc = float('nan')  
              u_prime.append(u_acc)
          return u_prime

      ​

      import math
      ​
      def convergents_sqrt2(n):
          """
          Calcule les n premiers convergents de la fraction continue de √2.
          Retourne une liste de tuples (p_n, q_n).
          """
          a = [1] + [2] * (n - 1)  # √2 = [1; 2, 2, 2, ...]
          p = [1, a[0]]  # p_{-1} = 1, p_0 = a_0
          q = [0, 1]     # q_{-1} = 0, q_0 = 1
      ​
          for i in range(1, n):
              p.append(a[i] * p[i] + p[i - 1])
              q.append(a[i] * q[i] + q[i - 1])
      ​
          return [(p[i], q[i]) for i in range(1, n + 1)]
      ​
      def comparer_avec_sqrt2(n):
          """
          Affiche les convergents de √2, leur valeur approchée, 
          et l'erreur absolue par rapport à math.sqrt(2).
          """
          sqrt2 = math.sqrt(2)
          convs = convergents_sqrt2(n)
      ​
          print("Convergents de √2 :\n")
          for i, (num, den) in enumerate(convs):
              approx = num / den
              erreur = abs(approx - sqrt2)
              print(f"n={i+1:2d} : {num}/{den} = {approx:.10f}  (erreur = {erreur:.2e})")
      ​
      n = 10
      comparer_avec_sqrt2(n)
